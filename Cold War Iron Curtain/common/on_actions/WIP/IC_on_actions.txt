on_actions = {
	on_annex = {
		effect = {
			if  = {
				limit = {
					any_owned_state = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }					
					}
				}
				every_owned_state = {
					limit = {
						is_island_state = no
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }
					}
					set_state_flag = active_insurgency
					set_state_flag = anti_colonial_insurgency
					set_variable = { insurgency_strength = 20 }
					add_state_modifier = {
						modifier = {
							local_resources = -0.20
							local_supplies = -0.30
							local_intel_to_enemies = 0.25
							local_factory_sabotage = 0.10
							local_non_core_manpower = -0.30
							attrition = 0.30
							local_org_regain = -0.25
						}
					}
				}
			}
		}
	}


	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if  = {
				limit = {
					any_owned_state = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }					
					}
				}
				every_owned_state = {
					limit = {
						is_core_of = FROM
						is_island_state = no
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }
					}
					set_state_flag = active_insurgency
					set_state_flag = anti_colonial_insurgency
					set_variable = { insurgency_strength = 20 }
					add_state_modifier = {
						modifier = {
							local_resources = -0.20
							local_supplies = -0.30
							local_intel_to_enemies = 0.25
							local_factory_sabotage = 0.10
							local_non_core_manpower = -0.30
							attrition = 0.30
							local_org_regain = -0.25
						}
					}
				}
			}
		}
	}
}
