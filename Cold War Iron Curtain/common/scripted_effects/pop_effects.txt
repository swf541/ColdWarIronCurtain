remove_pop_scale = {
	if = {
		limit = {
			FROM.FROM = { 
				state_population > 50000
				NOT = { state_population > 100000 }
			}				
		}
		FROM.FROM = {
			add_manpower = -5000
			
		}
		
	}
	if = {
		limit = {

			FROM.FROM = { 
				state_population > 100000
				NOT = { state_population > 1000000 }
			}				
		}
		FROM.FROM = {
			add_manpower = -10000
		}
		
	}
	if = {
		limit = {
			FROM.FROM = { 
				state_population > 1000000
				NOT = { state_population > 2000000 }
			}				
		}
		FROM.FROM = {
			add_manpower = -20000
		}
		
	}
	if = {
		limit = {
			FROM.FROM = { 
				state_population > 2000000
				NOT = { state_population > 5000000 }
			}				
		}
		FROM.FROM = {
			add_manpower = -50000
		}
		
	}
	if = {
		limit = {
			FROM.FROM = { 
				state_population > 5000000
			}		
		}
		FROM.FROM = {
			add_manpower = -500000
		}
		
	}
	if = {
		limit = {
			has_tech = atomic_weapons_armament_0
			FROM.FROM = { 
				state_population > 10000000
			}		
		}
		FROM.FROM = {
			add_manpower = -1000000
		}
		
	}
}

stronghold_drift = { #Ensures that the minimum population (before other calculations) follows the stronghold's religion by a percent equal to the stronghold's value. IE THE RELIGION CAN NEVER FALL BELOW THE STRONGHOLD STRENGTH% IN THIS STATE
	set_temp_variable = { strongholdTrigger = state_population_k }
	
	set_temp_variable = { strongholdTriggerReligionSpiritual = strongholdTrigger } #stronghold trigger = state pops
	multiply_temp_variable = { strongholdTriggerReligionSpiritual = ReligionSpiritualStronghold } #stronghold trigger = state pops x stronghold value (desired min)
	if = {
		limit = { AND = { check_variable = { ReligionSpiritualStronghold > 0 } check_variable = { ReligionSpiritual < strongholdTriggerReligionSpiritual } } } #Religious presence less than the stronghold religion's desired minimum
		set_temp_variable = { strongholdVar = state_population_k }
		multiply_temp_variable = { strongholdVar = ReligionSpiritualStronghold }
		set_variable = { ReligionSpiritualSafe = strongholdVar }
	}
	
	set_temp_variable = { strongholdTriggerReligionCultistSafe = strongholdTrigger }

		limit = { AND = { check_variable = { ReligionCultistSafeStronghold > 0 } check_variable = { ReligionCultistSafe < strongholdTriggerReligionCultistSafe } } }
