	name= SESSION_STATUS_INFLUENCE
	
	text= {
		trigger = {
			check_variable = { influence_countdown = 0}
		}
		localization_key = "No Diplomatic Mission Active"
	}
	
	text= {
		trigger = {
					AND= {
						 check_variable = { influence_countdown = 1}
						}
					}
				localization_key = "Diplomatic Mission Active"
	}
}
