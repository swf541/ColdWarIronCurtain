UAR_Kingdom_of_Yemen_decisions = {
	Spread_Pan_Arab_Propaganda_YEM = {

		icon = generic_prepare_civil_war

		available = {
			tag = EGY
		}
		
		complete_effect = {
			YEM = {
				add_popularity = {
					ideology = nationalist 
					popularity = 0.01
				}
				for_loop_effect = {
					end = influence_array^num
					value = v
					if = {
						limit = {
							check_variable = { influence_array^v = PREV }
						}
						add_to_variable = { influence_array_val^v = 100 }
						set_country_flag = found
					}
				}
				if = {
					limit = { NOT = { has_country_flag = found } }
						add_to_array = { influence_array = PREV.id }
						add_to_array = { influence_array_val = 100 }
				}
				clr_country_flag = found
				recalculate_influence = yes				
			}
		}

		cost = 100
		days_re_enable = 365

		fire_only_once = no
		selectable_mission = no

		ai_will_do = {
			factor = 0
		}
	}
	Integrate_YEM = {

		icon = generic_prepare_civil_war

		available = {
			tag = EGY
			YEM = { has_government = nationalist } 
			has_war = no
			custom_trigger_tooltip = {
				tooltip = Integration_UAR
				check_variable = { YEM.influence_array^0 = ROOT }
				check_variable = { YEM.influence_array_calc^0 > 0.74 }
			}
			YEM = {  }
		}
		
		complete_effect = {
			news_event = PRO_UAR_COUP.1
			annex_country = {
				target = YEM
				transfer_troops = yes
			}
			every_owned_state = {
				limit = { is_core_of = YEM }
				add_core_of = EGY
			}
		}

		cost = 0
		days_re_enable = 1625

		fire_only_once = no
		selectable_mission = no

		ai_will_do = {
			factor = 0
		}
	}
}
