		}

				tooltip = cannot_afford

		complete_effect = {
			subtract_from_variable = { currentMoney = 25 }

				limit = {

				full_upgrade_estimate_regen = yes
			}
			else_if = {
				limit = {
					any_owned_state = {
						OR = {

					}
					check_variable = { industrialization_budget > global.UPC_pop5 }
