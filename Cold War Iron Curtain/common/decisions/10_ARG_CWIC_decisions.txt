#ARGENTINIAN FACTION DECISIONS
ARG_faction_management_category = {
	ARG_befriend_trade_unions = {

		icon = generic_political_rally

		allowed = { 
			original_tag = ARG
		}

		available = {
			has_country_flag = ARG_factions_unlocked
			custom_trigger_tooltip = {
				tooltip = ARG_trigger_cgt_it_s_inconsistencial_or_indifferent_tt		
				OR = {
					check_variable = { ARG_trade_unions_loyalty = -2 } #-2 inconsintencial
					check_variable = { ARG_trade_unions_loyalty = 0 } #0 indifferent
				}
			}
		}

		cost = 300

		fire_only_once = yes

		days_remove = 100

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 5
				check_variable = { ARG_trade_unions_loyalty = 0 compare = less_than_or_equals }
			}
			modifier = {
				factor = 25
				check_variable = { ARG_trade_unions_loyalty = -1 compare = less_than_or_equals }
			}
		}

		visible = {
			always = yes
		}

		complete_effect = {
		}

		remove_effect = {
			effect_tooltip = { add_ideas = ARG_friendly_trade_unions }
			set_variable = { ARG_trade_unions_loyalty = 1 }
			hidden_effect = {
				ARG_political_instability_update_effect = yes
			}
		}

		modifier = {
			consumer_goods_factor = -0.05
		}
	}

	ARG_give_favor_to_cgt = {

		icon = generic_political_rally

		allowed = { 
			original_tag = ARG
		}

		available = {
			has_country_flag = ARG_factions_unlocked
			#2 loyal
			#1 friendly
			#0 indifferent
			#-1 hostile
			#-2 inconsintencial

			custom_trigger_tooltip = {
				tooltip = ARG_trigger_cgt_it_s_loyal_or_friendly_or_inconsistencial_or_indifferent_tt		
				OR = {
					check_variable = { ARG_trade_unions_loyalty = -2 } #-2 inconsintencial
					check_variable = { ARG_trade_unions_loyalty = 0 } #0 indifferent
					check_variable = { ARG_trade_unions_loyalty = 1 } #1 friendly
					check_variable = { ARG_trade_unions_loyalty = 2 } #2 loyal
				}
			}
		}

		cost = 250

		fire_only_once = yes

		days_remove = 200

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 5
				check_variable = { ARG_trade_unions_loyalty = 0 compare = less_than_or_equals }
			}
			modifier = {
				factor = 25
				check_variable = { ARG_trade_unions_loyalty = -1 compare = less_than_or_equals }
			}
		}

		visible = {
			always = yes
		}

		complete_effect = {
			
		}

		remove_effect = {
			effect_tooltip = { add_ideas = ARG_friendly_trade_unions }
			hidden_effect = {
				add_to_variable = { ARG_trade_unions_loyalty = 1 }
				ARG_political_instability_update_effect = yes
			}
		}
	}
}
