	id = thetroubles.0 # IRA meets
	title = "IRA High Command"
	desc = thetroubles.0.d 
	major = no 
	
	is_triggered_only = yes 
	
	hidden = yes
	immediate = {
		name = "Tiocfaidh ar la"
		activate_mission = Irish_Republican_Army
		IF = { has_token = token_informers
				activate_mission = anti_informer
		}
		IF = { has_idea = internment 
				add_to_variable = { IRA = 2 }
		}
		IF = {
			NOT = {
				has_operation_token = {
					tag = ROOT 
					token = token_informers
				}
				activate_mission = IRA_attack
			}
		}
		IF = {
			check_variable = { IRA < 5 }
			country_event = thetroubles.43
		}
	}
}
